// app/src/main/cpp/CmakeLists.txt
cmake_minimum_required(VERSION 3.22.1)
project(flux_android LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ── Path to your flux library source tree ────────────────────────────────────
set(FLUX_SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/flux")

# ── Build nanovg HERE at the top level so both flux and native-lib can
#    reference the same target by name.  flux/CMakeLists.txt must NOT
#    define its own nanovg target when built as a subdirectory.
# ─────────────────────────────────────────────────────────────────────────────
add_library(nanovg STATIC
        ${FLUX_SOURCE_DIR}/external/nanovg/src/nanovg.c
        ${FLUX_SOURCE_DIR}/external/nanovg/src/nanovg_gl_impl.c
)
set_source_files_properties(
        ${FLUX_SOURCE_DIR}/external/nanovg/src/nanovg.c
        PROPERTIES LANGUAGE C
)
target_include_directories(nanovg PUBLIC
        ${FLUX_SOURCE_DIR}/external/nanovg/src
)
# PRIVATE — only nanovg.c gets the implementation macro.
# flux and native-lib include nanovg_gl.h for declarations only.


# ── flux library (nanovg target already exists, flux/CMakeLists.txt
#    will find it via target_link_libraries rather than re-creating it)
add_subdirectory(${FLUX_SOURCE_DIR} flux_build)

# ── native-lib ────────────────────────────────────────────────────────────────
add_library(native-lib SHARED
        native-lib.cpp
        app.cpp
)

target_include_directories(native-lib PRIVATE
        ${FLUX_SOURCE_DIR}/include
        ${FLUX_SOURCE_DIR}/external/nanovg/src
        ${ANDROID_NDK}/sources/android/native_app_glue
)

target_compile_definitions(native-lib PRIVATE
        NANOVG_GLES2    # selects GLES2 declarations in nanovg_gl.h
        # NOT NANOVG_GLES2_IMPLEMENTATION — nanovg.c owns that
)

# nanovg first so nvgCreateGLES2 / nvgDeleteGLES2 resolve before the
# linker moves past the archive.
target_link_libraries(native-lib PRIVATE
        nanovg
        flux::flux
)

target_link_options(native-lib PRIVATE
        "-Wl,-u,ANativeActivity_onCreate"
)
